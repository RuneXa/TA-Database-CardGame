I have edited the answer to include all considerations in the comments. First though, make sure your script name is the same as the class name (e.g. public class DeckOfCards : MonoBehaviour -> script name should be DeckOfCards) : 

  1.  using UnityEngine;
  2.  using System.Collections;
  3.  using System.Collections.Generic;
  4.  
  5.  public class DeckOfCards : MonoBehaviour
  6.  {
  7.      public List<GameObject> deck = new List<GameObject>();
  8.  
  9.      private List<GameObject> cards = new List<GameObject>();
  10.      private List<GameObject> hand = new List<GameObject>();
  11.      private int cardsDealt = 0;
  12.      private bool showReset = false;
  13.  
  14.      void ResetDeck()
  15.      {
  16.          cardsDealt = 0;
  17.          for (int i = 0; i < hand.Count; i++) {
  18.              Destroy(hand[i]);
  19.          }
  20.          hand.Clear();
  21.          cards.Clear();
  22.          cards.AddRange(deck);
  23.          showReset = false;
  24.      }
  25.  
  26.      GameObject DealCard()
  27.      {
  28.          if(cards.Count == 0)
  29.          {
  30.              showReset = true;
  31.              return null;
  32.              //Alternatively to auto reset the deck:
  33.              //ResetDeck();
  34.          }
  35.  
  36.          int card = Random.Range(0, cards.Count - 1);
  37.          GameObject go = GameObject.Instantiate(cards[card]) as GameObject;
  38.          cards.RemoveAt(card);
  39.          
  40.          if(cards.Count == 0) {
  41.              showReset = true;
  42.          }
  43.          
  44.          return go;
  45.      }
  46.  
  47.      void Start()
  48.      {
  49.          ResetDeck();
  50.      }
  51.  
  52.      void GameOver()
  53.      {
  54.          cardsDealt = 0;
  55.          for (int v = 0; v < hand.Count; v++) {
  56.              Destroy(hand[v]);
  57.          }
  58.          hand.Clear();
  59.          cards.Clear();
  60.          cards.AddRange(deck);
  61.      }
  62.  
  63.      void OnGUI()
  64.      {
  65.          if (!showReset) {
  66.              // Deal button
  67.              if (GUI.Button(new Rect(10, 10, 100, 20), "Deal"))
  68.              {
  69.                  MoveDealtCard();
  70.              }
  71.          }
  72.          else {
  73.              // Reset button
  74.              if (GUI.Button(new Rect(10, 10, 100, 20), "Reset"))
  75.              {
  76.                  ResetDeck();
  77.              }
  78.          }
  79.          // GameOver button
  80.          if (GUI.Button(new Rect(Screen.width - 110, 10, 100, 20), "GameOver"))
  81.          {
  82.              GameOver();
  83.          }
  84.      }
  85.  
  86.      void MoveDealtCard()
  87.      {
  88.          GameObject newCard = DealCard();
  89.          // check card is null or not
  90.          if (newCard == null) {
  91.              Debug.Log("Out of Cards");
  92.              showReset = true;
  93.              return;
  94.          }
  95.          
  96.          //newCard.transform.position = Vector3.zero;
  97.          newCard.transform.position = new Vector3((float)cardsDealt / 4, (float)cardsDealt / -4, (float)cardsDealt / -4); // place card 1/4 up on all axis from last
  98.          hand.Add(newCard); // add card to hand
  99.          cardsDealt ++;
  100.      }
  101.  }
  102.  

Now here's where you drag and drop your card prefabs. Look in the inspector and you'll see Deck : Size : 0 .

Now change the 0 to 52. you should have the second screenshot.

You could create a script with a list of prefabs, use the inspector to make it 52 long and then drop the prefabs into each slot. That gives you 52 cards that you can access.

Your then need to make a copy of the deck and deal out the cards one at a time in random order, removing it from the copied deck.

This script might help:

  1.  #pragma strict
  2.  
  3.  import System.Collections.Generic;
  4.  
  5.  var deck : List.<GameObject>;
  6.  
  7.  private var cards : List.<GameObject> = new List.<GameObject>();
  8.  
  9.  function ResetDeck()
  10.  {
  11.      cards.Clear();
  12.      cards.AddRange(deck);
  13.  }
  14.  
  15.  function DealCard() : GameObject
  16.  {
  17.      if(cards.Count==0)
  18.      {
  19.          return null;
  20.          //Alternatively to auto reset the deck:
  21.          //ResetDeck();
  22.      }
  23.      
  24.      var card : int = Random.Range(0, cards.Count-1);
  25.      var go : GameObject = GameObject.Instantiate(cards[card]) as GameObject;
  26.      cards.RemoveAt(card);
  27.      return go;
  28.  }
  29.  
  30.  function Start()
  31.  {
  32.      ResetDeck();
  33.  }
  34.  
  35.  function RemainingCardCount() 
  36.  {
  37.      return cards.Count;
  38.  }
  39.  

When you call DealCard it will create a new GameObject from the prefab - you'd then need to position it etc. After 52 cards it would start returning "null" or you could check the RemainingCardCount(). In the script you could delete the return null; line and uncomment the ResetDeck() line to have it automatically start again.

You would probably want to call this in an OnGUI function that would show your button and then position the created game object according to the camera you have and where you want the item to appear. A simple OnGUI might look like this:

  1.  function OnGUI()
  2.  {
  3.      if(GUILayout.Button("Deal"))
  4.      {
  5.          var newCard : GameObject = DealCard();
  6.          newCard.transform.position = Vector3.zero; //Replace with your location
  7.      }
  8.  }

Ok, I am guessing this help has stopped because you don't quite understand : "attach the script to a GameObject and view it in the inspector. When you expand Deck it will say 0 items, change this to 52 and you will get 52 slots. Drag the cards into those slots". That's ok, list's and arrays are hard to get your head around when starting.

The advice you have been given is to Forget putting all your cards into separate Transforms in the Inspector, and put then into the List deck in the Inspector. This is the best method by far, soo much extra typing doing it your way.

However , if you want to write some code that populates List deck, I can help. But first you need to change your typecasting . What this means is you are putting your cards into a variable that is typecast as a Transform. e.g. public Transform Diamond1; Now if you look at the List it is typecast as GameObject. e.g. List deck = new List();

Can you see Transform and GameObject? To put Diamond1 into (the List) deck, the Diamond has to be typecast as GameObject , or the deck has to be typecast as Transform. I suggest you store the GameObject, as you can access all the components from this typecast (inc transform).

Here is the Long method for populating the list. You'll need to do the other suits (just the same as clubs): 

  1.  using UnityEngine;
  2.  using System.Collections;
  3.  
  4.  public class DeckAndCards : MonoBehaviour
  5.  {
  6.      List<GameObject> deck = new List<GameObject>();
  7.  
  8.      private List<GameObject> cards = new List<GameObject>();
  9.      
  10.      public GameObject Club1;
  11.      public GameObject Club2;
  12.      public GameObject Club3;
  13.      public GameObject Club4;
  14.      public GameObject Club5;
  15.      public GameObject Club6;
  16.      public GameObject Club7;
  17.      public GameObject Club8;
  18.      public GameObject Club9;
  19.      public GameObject Club10;
  20.      public GameObject Club11;
  21.      public GameObject Club12;
  22.      public GameObject Club13; // you can do the other suits!
  23.  
  24.      void LoadDeckList()
  25.      {
  26.          deck.Add(Club1); 
  27.          deck.Add(Club2); 
  28.          deck.Add(Club3); 
  29.          deck.Add(Club4);
  30.          deck.Add(Club5);
  31.          deck.Add(Club6);
  32.          deck.Add(Club7);
  33.          deck.Add(Club8);
  34.          deck.Add(Club9);
  35.          deck.Add(Club10);
  36.          deck.Add(Club11);
  37.          deck.Add(Club12);
  38.          deck.Add(Club13);
  39.          
  40.          Debug.Log(" List deck = " + deck); // this should show your List deck in the console, and show that the cards have been added
  41.      }
  42.

  44.      {
  45.          // NOW load the List deck first
  46.          deck.Clear();
